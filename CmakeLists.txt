cmake_minimum_required(VERSION 3.10)
project(SDK)

# Enable debugging symbols
set(CMAKE_BUILD_TYPE Debug)
set(TARGET_NAME SDK)

# 启用C++11
add_compile_options(-std=c++11)
# 解决节数超过对象文件格式限制
add_compile_options(-bigobj)

file(GLOB_RECURSE SRC_FILE "*.cpp" "*.cxx" "*.cc" "*.C" "*.c++" "*.h" "*.hpp" "*.H" "*.hxx")

add_executable(SDK ./main.cpp)

# Add compiler flags for debugging
set(CMAKE_CXX_FLAGS_DEBUG "${CMAKE_CXX_FLAGS_DEBUG} -g")

# Include directories for FileSystem and JsonObject
target_include_directories(${TARGET_NAME} PUBLIC .)
target_include_directories(${TARGET_NAME} PUBLIC ${CMAKE_SOURCE_DIR}/FileSystem)
target_include_directories(${TARGET_NAME} PUBLIC ${CMAKE_SOURCE_DIR}/jsonObject)

# Link against the necessary libraries
target_link_libraries(SDK PRIVATE 
    /path/to/FileSystem/lib/libFileSystem.a
    /path/to/JsonObject/lib/libJsonObject.a
)
